class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        sort(nums.begin(), nums.end());
        int count = 0;
        for(int i=0; i<nums.size(); i++){
            if(nums[i] == val){
                nums[i] = INT_MAX;
                count++;
            }
        }
        sort(nums.begin(), nums.end());
        return nums.size()-count;
    }
};
